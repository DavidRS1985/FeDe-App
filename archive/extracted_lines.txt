34:   function calculateEaster(year) {
35:     const a = year % 19, b = Math.floor(year / 100), c = year % 100, d = Math.floor(b / 4), e = b % 4,
36:       f = Math.floor((b + 8) / 25), g = Math.floor((b - f + 1) / 3), h = (19 * a + b - d - g + 15) % 30,
37:       i = Math.floor(c / 4), k = c % 4, l = (32 + 2 * e + 2 * i - h - k) % 7,
38:       m = Math.floor((a + 11 * h + 22 * l) / 451), month = Math.floor((h + l - 7 * m + 114) / 31),
39:       day = ((h + l - 7 * m + 114) % 31) + 1;
40:     return new Date(year, month - 1, day);
41:   }

525:     function onConfigLoaded(result) {
526:         // Limpiar TODOS los timeouts al recibir respuesta del servidor
527:         if (window._masterBootTimeout) { clearTimeout(window._masterBootTimeout); window._masterBootTimeout = null; }
528:         if (window.loaderTimeout) { clearTimeout(window.loaderTimeout); window.loaderTimeout = null; }

534:             return;
535:         }

539:             
540:             // Actualizar nombre de DB
541:             if (ssName) { 
542:                 dbName = isOfflineMode ? `${ssName} (Offline)` : ssName;
543:                 updateDBLabel(); 
544:             }
545: 
546:             // Cargar datos - Funciona en línea o sin conexión
547:             if (config) {
548:                 if (config.cats) CATS = config.cats;
549:                 if (config.favs) FAVS = config.favs;
550:                 if (config.units) units = config.units;

562:             }));

564:             // Mostrar estado

566:                 console.warn('⚠️ Modo Offline: ', error);

614:         if (window._masterBootTimeout) {

616:             window._masterBootTimeout = null;
617:         }

619:         if (loadingInterval) { clearInterval(loadingInterval); loadingInterval = null; }

621:         const loader = document.getElementById('loader');

623:         
624:         // Si ya está oculto, no hacer nada más
625:         if (loader.style.display === 'none') return;

637:         // Ocultar sin animación en timeouts

641:             loader.style.pointerEvents = 'none';
